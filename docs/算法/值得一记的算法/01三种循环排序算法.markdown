数据结构里面有一种bool类型的冒泡排序,很巧妙

``` c
void BubbleSort(int score[], int n) {
	//12345  12 23 34 45 i=0---23 34 45 i=1
	int i, j, temp;
	for (i = 0; i < n - 1; i++) {
	//大循环n-1次,表示冒了几个泡；执行到n-2的那个元素
		for (j = 1; j < n - i; j++) {
		//每一个小循环执行n-i-1次,从1开始,所以j<n-i
			if (score[j] < score[j - 1]) {
				temp = score[j];
				score[j] = score[j - 1];
				score[j - 1] = temp;
			}//冒泡排序即比较后交换相邻元素
		}
	}
}
```

``` c
void Changesort(int score[], int n) {
	int i, j, temp;
	for (i = 0; i < n - 1; i++) {
	//外层因为最后一个不用,所以执行元素到n-2；
		for (j = i + 1; j < n; j++) {
		//每一次j=i+1开始;一直进行到n-1的下标元素
			if (score[j] < score[i]) {
				temp = score[j];
				score[j] = score[i];
				score[i] = temp;
			}//第一遍用第一个和剩下所有的比；把小的换到第一位
		}
	}
}
```

``` c
void SelectionSort(int score[], int n) {
	int i,j,k,temp;
	for (i = 0; i < n - 1; i++) {
		k = i;//记录了开始时的位置
		for (j = i + 1; j < n; j++) {
		//每一次j=i+1开始;一直进行到n-1的下标元素
			if (score[j] < score[k]) {
				k = j;//记录下来最小的那个的下标
			}
		}
		if (k != i) {
			temp = score[k];
			score[k] = score[i];
			score[i] = temp;
		}
	}
}
```
